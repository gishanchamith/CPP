üåâ Kruskal's Algorithm (MST, O(E log E))

struct DSU {
    vector p;
    DSU(int n): p(n) { iota(p.begin(), p.end(), 0); }
    int find(int x) { return x == p[x] ? x : p[x] = find(p[x]); }
    bool unite(int x, int y) {
        x = find(x); y = find(y);
        if (x == y) return false;
        p[y] = x; return true;
    }
};

struct Edge { int u, v, w; };
vector edges;
int kruskal(int n) {
    sort(edges.begin(), edges.end(), [](Edge a, Edge b){ return a.w < b.w; });
    DSU dsu(n); int mst = 0;
    for (auto& e : edges)
        if (dsu.unite(e.u, e.v)) mst += e.w;
    return mst;
}

int prim(int n, vector>> &adj) {
    vector vis(n, false);
    priority_queue, vector>, greater<>> pq;
    pq.push({0, 0}); int cost = 0;
    while (!pq.empty()) {
        auto [w,u] = pq.top(); pq.pop();
        if (vis[u]) continue;
        vis[u] = true; cost += w;
        for (auto &[v, wt] : adj[u])
            if (!vis[v]) pq.push({wt, v});
    }
    return cost;
}

‚ö° SPFA (Shortest Path Faster Algorithm)

vector spfa(int n, int src, vector>> &adj) {
    vector d(n, LLONG_MAX); d[src] = 0;
    vector inq(n, false); queue q; q.push(src); inq[src]=1;
    while (!q.empty()) {
        int u = q.front(); q.pop(); inq[u]=0;
        for (auto &[v,w] : adj[u])
            if (d[v] > d[u]+w) {
                d[v]=d[u]+w;
                if (!inq[v]) q.push(v), inq[v]=1;
            }
    }
    return d;
}
