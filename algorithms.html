🔃 2. Sorting Algorithms

Bubble Sort

for (int i = 0; i < n; i++) {
  for (int j = 0; j < n - i - 1; j++) {
    if (arr[j] > arr[j+1])
      swap(arr[j], arr[j+1]);
  }
}
  • Time: O(n²), Stable: Yes

Insertion Sort

for (int i = 1; i < n; i++) {
  int key = arr[i], j = i - 1;
  while (j >= 0 && arr[j] > key) {
    arr[j + 1] = arr[j];
    j--;
  }
  arr[j + 1] = key;
}
  • Time: O(n²) worst, O(n) best, Stable: Yes

Merge Sort (Divide & Conquer)

void mergeSort(int arr[], int l, int r) {
  if (l < r) {
    int m = l + (r - l) / 2;
    mergeSort(arr, l, m);
    mergeSort(arr, m + 1, r);
    merge(arr, l, m, r);
  }
}
  • Time: O(n log n), Stable: Yes

🔁 3. Recursion and Divide & Conquer

Recursion Example

int factorial(int n) {
  if (n <= 1) return 1;
  return n * factorial(n - 1);
}

Divide & Conquer: Binary Search

int binarySearch(int arr[], int low, int high, int x) {
  if (low > high) return -1;
  int mid = (low + high) / 2;
  if (arr[mid] == x) return mid;
  if (arr[mid] > x)
    return binarySearch(arr, low, mid - 1, x);
  else
    return binarySearch(arr, mid + 1, high, x);
}

🌳 6. Binary Search Trees (BST)

  • Property: Left < Node, Right > Node
  • Traversals: Inorder, Preorder, Postorder
Node* insert(Node* root, int key) {
  if (!root) return new Node(key);
  if (key < root->val) root->left = insert(root->left, key);
  else root->right = insert(root->right, key);
  return root;
}

🏗 7. Heaps and Heap Sort

🧠 12. Algorithm Design Techniques

  • Brute Force: Try everything
  • Greedy: Best choice each step
  • Dynamic Programming: Solve overlapping subproblems
int fib(int n) {
  int dp[n+1];
  dp[0]=0, dp[1]=1;
  for (int i=2; i<=n; i++)
    dp[i] = dp[i-1] + dp[i-2];
  return dp[n];
}
